171 Allocator *Allocator::create(
172  CephContext* cct,
173  std::string_view type,
174  int64_t size,
175  int64_t block_size,
176  int64_t zone_size,
177  int64_t first_sequential_zone,
178  std::string_view name)
179 {
180  Allocator* alloc = nullptr;
181  if (type == "stupid") {
182  alloc = new StupidAllocator(cct, size, block_size, name);
183  } else if (type == "bitmap") {
184  alloc = new BitmapAllocator(cct, size, block_size, name);
185  } else if (type == "avl") {
186  return new AvlAllocator(cct, size, block_size, name);
187  } else if (type == "btree") {
188  return new BtreeAllocator(cct, size, block_size, name);
189  } else if (type == "hybrid") {
190  return new HybridAllocator(cct, size, block_size,
191  cct->_conf.get_val<uint64_t>("bluestore_hybrid_alloc_mem_cap"),
192  name);
193 #ifdef HAVE_LIBZBD
194  } else if (type == "zoned") {
195  return new ZonedAllocator(cct, size, block_size, zone_size, first_sequential_zone,
196  name);
197 #endif
198  }
199  if (alloc == nullptr) {
200  lderr(cct) << "Allocator::" << __func__ << " unknown alloc type "
201  << type << dendl;
202  }
203  return alloc;
204 }

231 double Allocator::get_fragmentation_score()
232 {
233  // this value represents how much worth is 2X bytes in one chunk then in X + X bytes
234  static const double double_size_worth_small = 1.2;
235  // chunks larger then 128MB are large enough that should be counted without penalty
236  static const double double_size_worth_huge = 1;
237  static const size_t small_chunk_p2 = 20; // 1MB
238  static const size_t huge_chunk_p2 = 27; // 128MB
239  // for chunks 1MB - 128MB penalty coeffs are linearly weighted 1.2 (at small) ... 1 (at huge)
240  static std::vector<double> scales{1};
241  double score_sum = 0;
242  size_t sum = 0;
243 
244  auto get_score = [&](size_t v) -> double {
245  size_t sc = sizeof(v) * 8 - std::countl_zero(v) - 1; //assign to grade depending on log2(len)
246  while (scales.size() <= sc + 1) {
247  //unlikely expand scales vector
248  auto ss = scales.size();
249  double scale = double_size_worth_small;
250  if (ss >= huge_chunk_p2) {
251  scale = double_size_worth_huge;
252  } else if (ss > small_chunk_p2) {
253  // linear decrease 1.2 ... 1
254  scale = (double_size_worth_huge * (ss - small_chunk_p2) + double_size_worth_small * (huge_chunk_p2 - ss)) /
255  (huge_chunk_p2 - small_chunk_p2);
256  }
257  scales.push_back(scales[scales.size() - 1] * scale);
258  }
259  size_t sc_shifted = size_t(1) << sc;
260  double x = double(v - sc_shifted) / sc_shifted; //x is <0,1) in its scale grade
261  // linear extrapolation in its scale grade
262  double score = (sc_shifted ) * scales[sc] * (1-x) +
263  (sc_shifted * 2) * scales[sc+1] * x;
264  return score;
265  };
266 
267  auto iterated_allocation = [&](size_t off, size_t len) {
268  ceph_assert(len > 0);
269  score_sum += get_score(len);
270  sum += len;
271  };
272  foreach(iterated_allocation);
273 
274  double ideal = get_score(sum);
275  double terrible = (sum / block_size) * get_score(block_size);
276  return (ideal - score_sum) / (ideal - terrible);
277 }

279 void Allocator::build_free_state_histogram(
280  size_t alloc_unit, Allocator::FreeStateHistogram& hist)
281 {
282  auto num_buckets = hist.size();
283  ceph_assert(num_buckets);
284 
285  auto base = free_state_hist_bucket::base;
286  auto base_bits = free_state_hist_bucket::base_bits;
287  auto mux = free_state_hist_bucket::mux;
288  // maximum chunk size we track,
289  // provided by the bucket before the last one
290  size_t max =
291  free_state_hist_bucket::get_max(num_buckets - 2, num_buckets);
292 
293  auto iterated_allocation = [&](size_t off, size_t len) {
294  size_t idx;
295  if (len <= base) {
296  idx = 0;
297  } else if (len > max) {
298  idx = num_buckets - 1;
299  } else {
300  size_t most_bit = cbits(uint64_t(len-1)) - 1;
301  idx = 1 + ((most_bit - base_bits) / mux);
302  }
303  ceph_assert(idx < num_buckets);
304  ++hist[idx].total;
305 
306  // now calculate the bucket for the chunk after alignment,
307  // resulting chunks shorter than alloc_unit are discarded
308  auto delta = p2roundup(off, alloc_unit) - off;
309  if (len >= delta + alloc_unit) {
310  len -= delta;
311  if (len <= base) {
312  idx = 0;
313  } else if (len > max) {
314  idx = num_buckets - 1;
315  } else {
316  size_t most_bit = cbits(uint64_t(len-1)) - 1;
317  idx = 1 + ((most_bit - base_bits) / mux);
318  }
319  ++hist[idx].aligned;
320  hist[idx].alloc_units += len / alloc_unit;
321  }
322  };
323 
324  foreach(iterated_allocation);
325 }
