42  const range_value_t& rhs) const noexcept {
43  if (lhs.size < rhs.size) {
44  return -1;
45  } else if (lhs.size > rhs.size) {
46  return 1;
47  }
48  if (lhs.start < rhs.start) {
49  return -1;

51  return 1;
52  } else {
53  return 0;
54  }
