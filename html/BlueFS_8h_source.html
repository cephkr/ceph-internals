230  FileWriter(FileRef f)
231  : file(std::move(f)),
232  buffer_appender(buffer.get_page_aligned_appender(
233  g_conf()->bluefs_alloc_size / CEPH_PAGE_SIZE)) {
234  ++file->num_writers;
235  iocv.fill(nullptr);
236  dirty_devs.fill(false);

238  write_hint = WRITE_LIFE_MEDIUM;

242  ~FileWriter() {
243  --file->num_writers;
244  }

246  // note: BlueRocksEnv uses this append exclusively, so it's safe
247  // to use buffer_appender exclusively here (e.g., its notion of
248  // offset will remain accurate).
249  void append(const char *buf, size_t len) {
250  uint64_t l0 = get_buffer_length();
251  ceph_assert(l0 + len <= std::numeric_limits<unsigned>::max());
252  buffer_appender.append(buf, len);
253  }
254 
255  void append(const std::byte *buf, size_t len) {
256  // allow callers to use byte type instead of char* as we simply pass byte array
257  append((const char*)buf, len);
258  }
259 
260  // note: used internally only, for ino 1 or 0.
261  void append(ceph::buffer::list& bl) {
262  uint64_t l0 = get_buffer_length();
263  ceph_assert(l0 + bl.length() <= std::numeric_limits<unsigned>::max());
264  buffer.claim_append(bl);
265  }
266 
267  void append_zero(size_t len) {
268  uint64_t l0 = get_buffer_length();
269  ceph_assert(l0 + len <= std::numeric_limits<unsigned>::max());
270  buffer_appender.append_zero(len);
271  }
272 
273  uint64_t get_effective_write_pos() {
274  return pos + buffer.length();
275  }

280 
281  uint64_t bl_off = 0;
282  ceph::buffer::list bl;
283  uint64_t pos = 0;
284  uint64_t max_prefetch;
285 
286  explicit FileReaderBuffer(uint64_t mpf)
287  : max_prefetch(mpf) {}
288 
289  uint64_t get_buf_end() const {
290  return bl_off + bl.length();
291  }
292  uint64_t get_buf_remaining(uint64_t p) const {
293  if (p >= bl_off && p < bl_off + bl.length())
294  return bl_off + bl.length() - p;
295  return 0;
296  }
297 
298  void skip(size_t n) {
299  pos += n;
300  }
301 
302  // For the sake of simplicity, we invalidate completed rather than
303  // for the provided extent
304  void invalidate_cache(uint64_t offset, uint64_t length) {
305  if (offset >= bl_off && offset < get_buf_end()) {
306  bl.clear();
307  bl_off = 0;
308  }
309  }
