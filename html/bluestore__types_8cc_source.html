114 void bluestore_extent_ref_map_t::_check() const
115 {
116  uint64_t pos = 0;
117  unsigned refs = 0;
118  for (const auto &p : ref_map) {
119  if (p.first < pos)
120  ceph_abort_msg("overlap");
121  if (p.first == pos && p.second.refs == refs)
122  ceph_abort_msg("unmerged");
123  pos = p.first + p.second.length;
124  refs = p.second.refs;
125  }
126 }

128 void bluestore_extent_ref_map_t::_maybe_merge_left(
129  map<uint64_t,record_t>::iterator& p)
130 {
131  if (p == ref_map.begin())
132  return;
133  auto q = p;
134  --q;
135  if (q->second.refs == p->second.refs &&
136  q->first + q->second.length == p->first) {
137  q->second.length += p->second.length;
138  ref_map.erase(p);
139  p = q;
140  }
141 }

143 void bluestore_extent_ref_map_t::get(uint64_t offset, uint32_t length)
144 {
145  auto p = ref_map.lower_bound(offset);
146  if (p != ref_map.begin()) {
147  --p;
148  if (p->first + p->second.length <= offset) {
149  ++p;
150  }
151  }
152  while (length > 0) {
153  if (p == ref_map.end()) {
154  // nothing after offset; add the whole thing.
155  p = ref_map.insert(
156  map<uint64_t,record_t>::value_type(offset, record_t(length, 1))).first;
157  break;
158  }
159  if (p->first > offset) {
160  // gap
161  uint64_t newlen = std::min<uint64_t>(p->first - offset, length);
162  p = ref_map.insert(
163  map<uint64_t,record_t>::value_type(offset,
164  record_t(newlen, 1))).first;
165  offset += newlen;
166  length -= newlen;
167  _maybe_merge_left(p);
168  ++p;
169  continue;
170  }
171  if (p->first < offset) {
172  // split off the portion before offset
173  ceph_assert(p->first + p->second.length > offset);
174  uint64_t left = p->first + p->second.length - offset;
175  p->second.length = offset - p->first;
176  p = ref_map.insert(map<uint64_t,record_t>::value_type(
177  offset, record_t(left, p->second.refs))).first;
178  // continue below
179  }
180  ceph_assert(p->first == offset);
181  if (length < p->second.length) {
182  ref_map.insert(make_pair(offset + length,
183  record_t(p->second.length - length,
184  p->second.refs)));
185  p->second.length = length;
186  ++p->second.refs;
187  break;
188  }
189  ++p->second.refs;
190  offset += p->second.length;
191  length -= p->second.length;
192  _maybe_merge_left(p);
193  ++p;
194  }
195  if (p != ref_map.end())
196  _maybe_merge_left(p);
197  //_check();
198 }

200 void bluestore_extent_ref_map_t::put(
201  uint64_t offset, uint32_t length,
202  PExtentVector *release,
203  bool *maybe_unshared)
204 {
205  //NB: existing entries in 'release' container must be preserved!
206  bool unshared = true;
207  auto p = ref_map.lower_bound(offset);
208  if (p == ref_map.end() || p->first > offset) {
209  if (p == ref_map.begin()) {
210  ceph_abort_msg("put on missing extent (nothing before)");
211  }
212  --p;
213  if (p->first + p->second.length <= offset) {
214  ceph_abort_msg("put on missing extent (gap)");
215  }
216  }
217  if (p->first < offset) {
218  uint64_t left = p->first + p->second.length - offset;
219  p->second.length = offset - p->first;
220  if (p->second.refs != 1) {
221  unshared = false;
222  }
223  p = ref_map.insert(map<uint64_t,record_t>::value_type(
224  offset, record_t(left, p->second.refs))).first;
225  }
226  while (length > 0) {
227  ceph_assert(p->first == offset);
228  if (length < p->second.length) {
229  if (p->second.refs != 1) {
230  unshared = false;
231  }
232  ref_map.insert(make_pair(offset + length,
233  record_t(p->second.length - length,
234  p->second.refs)));
235  if (p->second.refs > 1) {
236  p->second.length = length;
237  --p->second.refs;
238  if (p->second.refs != 1) {
239  unshared = false;
240  }
241  _maybe_merge_left(p);
242  } else {
243  if (release)
244  release->push_back(bluestore_pextent_t(p->first, length));
245  ref_map.erase(p);
246  }
247  goto out;
248  }
249  offset += p->second.length;
250  length -= p->second.length;
251  if (p->second.refs > 1) {
252  --p->second.refs;
253  if (p->second.refs != 1) {
254  unshared = false;
255  }
256  _maybe_merge_left(p);
257  ++p;
258  } else {
259  if (release)
260  release->push_back(bluestore_pextent_t(p->first, p->second.length));
261  ref_map.erase(p++);
262  }
263  }
264  if (p != ref_map.end())
265  _maybe_merge_left(p);
266  //_check();
267 out:
268  if (maybe_unshared) {
269  if (unshared) {
270  // we haven't seen a ref != 1 yet; check the whole map.
271  for (auto& p : ref_map) {
272  if (p.second.refs != 1) {
273  unshared = false;
274  break;
275  }
276  }
277  }
278  *maybe_unshared = unshared;
279  }
280 }

289 bluestore_extent_ref_map_t::debug_len_cnt bluestore_extent_ref_map_t::debug_peek(uint64_t offset) const
290 {
291  // locate offset
292  auto p = ref_map.lower_bound(offset);
293  if (p != ref_map.end() && p->first == offset) {
294  // direct request for us
295  return {p->second.length, p->second.refs};
296  }
297  if (p != ref_map.begin()) {
298  --p;
299  if (p->first + p->second.length <= offset) {
300  // nah, it ends too soon, we landed in a hole
301  ++p;
302  if (p != ref_map.end()) {
303  // there is a region after
304  return {uint32_t(p->first - offset), 0};
305  } else {
306  // nothing after
307  return {0, 0};
308  }
309  } else {
310  // we're in the range
311  return {uint32_t((p->first + p->second.length) - offset), p->second.refs};
312  }
313  } else {
314  return {uint32_t(p->first - offset), 0};
315  }
316 }

318 bool bluestore_extent_ref_map_t::contains(uint64_t offset, uint32_t length) const
319 {
320  auto p = ref_map.lower_bound(offset);
321  if (p == ref_map.end() || p->first > offset) {
322  if (p == ref_map.begin()) {
323  return false; // nothing before
324  }
325  --p;
326  if (p->first + p->second.length <= offset) {
327  return false; // gap
328  }
329  }
330  while (length > 0) {
331  if (p == ref_map.end())
332  return false;
333  if (p->first > offset)
334  return false;
335  if (p->first + p->second.length >= offset + length)
336  return true;
337  uint64_t overlap = p->first + p->second.length - offset;
338  offset += overlap;
339  length -= overlap;
340  ++p;
341  }
342  return true;
343 }

348 {

351  --p;

354  }

357  return false;

472 void bluestore_blob_use_tracker_t::get(
473  uint32_t offset, uint32_t length)
474 {
475  ceph_assert(au_size);
476  if (!num_au) {
477  total_bytes += length;
478  } else {
479  auto end = offset + length;
480  if (end / au_size >= num_au) {
481  add_tail(end, au_size);
482  }
483  while (offset < end) {
484  auto phase = offset % au_size;
485  bytes_per_au[offset / au_size] +=
486  std::min(au_size - phase, end - offset);
487  offset += (phase ? au_size - phase : au_size);
488  }
489  }
490 }

492 bool bluestore_blob_use_tracker_t::put(
493  uint32_t offset, uint32_t length,
494  PExtentVector *release_units)
495 {
496  ceph_assert(au_size);
497  if (release_units) {
498  release_units->clear();
499  }
500  bool maybe_empty = true;
501  if (!num_au) {
502  ceph_assert(total_bytes >= length);
503  total_bytes -= length;
504  } else {
505  auto end = offset + length;
506  uint64_t next_offs = 0;
507  while (offset < end) {
508  auto phase = offset % au_size;
509  size_t pos = offset / au_size;
510  auto diff = std::min(au_size - phase, end - offset);
511  ceph_assert(diff <= bytes_per_au[pos]);
512  bytes_per_au[pos] -= diff;
513  offset += (phase ? au_size - phase : au_size);
514  if (bytes_per_au[pos] == 0) {
515  if (release_units) {
516  if (release_units->empty() || next_offs != pos * au_size) {
517  release_units->emplace_back(pos * au_size, au_size);
518  next_offs = pos * au_size;
519  } else {
520  release_units->back().length += au_size;
521  }
522  next_offs += au_size;
523  }
524  } else {
525  maybe_empty = false; // micro optimization detecting we aren't empty
526  // even in the affected extent
527  }
528  }
529  }
530  bool empty = maybe_empty ? !is_not_empty() : false;
531  if (empty && release_units) {
532  release_units->clear();
533  }
534  return empty;
535 }

549 void bluestore_blob_use_tracker_t::split(
550  uint32_t blob_offset,
551  bluestore_blob_use_tracker_t* r)
552 {
553  ceph_assert(au_size);
554  ceph_assert(can_split());
555  ceph_assert(can_split_at(blob_offset));
556  ceph_assert(r->is_empty());
557 
558  uint32_t new_num_au = blob_offset / au_size;
559  r->init( (num_au - new_num_au) * au_size, au_size);
560 
561  for (auto i = new_num_au; i < num_au; i++) {
562  r->get((i - new_num_au) * au_size, bytes_per_au[i]);
563  bytes_per_au[i] = 0;
564  }
565  if (new_num_au == 0) {
566  clear();
567  } else if (new_num_au == 1) {
568  uint32_t tmp = bytes_per_au[0];
569  uint32_t _au_size = au_size;
570  clear();
571  au_size = _au_size;
572  total_bytes = tmp;
573  } else {
574  num_au = new_num_au;
575  }
576 }

591 bool bluestore_blob_use_tracker_t::equal(
592  const bluestore_blob_use_tracker_t& other) const
593 {
594  if (!num_au && !other.num_au) {
595  return total_bytes == other.total_bytes && au_size == other.au_size;
596  } else if (num_au && other.num_au) {
597  if (num_au != other.num_au || au_size != other.au_size) {
598  return false;
599  }
600  for (size_t i = 0; i < num_au; i++) {
601  if (bytes_per_au[i] != other.bytes_per_au[i]) {
602  return false;
603  }
604  }
605  return true;
606  }
607 
608  uint32_t n = num_au ? num_au : other.num_au;
609  uint32_t referenced =
610  num_au ? other.get_referenced_bytes() : get_referenced_bytes();
611  auto bytes_per_au_tmp = num_au ? bytes_per_au : other.bytes_per_au;
612  uint32_t my_referenced = 0;
613  for (size_t i = 0; i < n; i++) {
614  my_referenced += bytes_per_au_tmp[i];
615  if (my_referenced > referenced) {
616  return false;
617  }
618  }
619  return my_referenced == referenced;
620 }

814 void bluestore_blob_t::calc_csum(uint64_t b_off, const bufferlist& bl)
815 {
816  switch (csum_type) {
817  case Checksummer::CSUM_XXHASH32:
818  Checksummer::calculate<Checksummer::xxhash32>(
819  get_csum_chunk_size(), b_off, bl.length(), bl, &csum_data);
820  break;
821  case Checksummer::CSUM_XXHASH64:
822  Checksummer::calculate<Checksummer::xxhash64>(
823  get_csum_chunk_size(), b_off, bl.length(), bl, &csum_data);
824  break;;
825  case Checksummer::CSUM_CRC32C:
826  Checksummer::calculate<Checksummer::crc32c>(
827  get_csum_chunk_size(), b_off, bl.length(), bl, &csum_data);
828  break;
829  case Checksummer::CSUM_CRC32C_16:
830  Checksummer::calculate<Checksummer::crc32c_16>(
831  get_csum_chunk_size(), b_off, bl.length(), bl, &csum_data);
832  break;
833  case Checksummer::CSUM_CRC32C_8:
834  Checksummer::calculate<Checksummer::crc32c_8>(
835  get_csum_chunk_size(), b_off, bl.length(), bl, &csum_data);
836  break;
837  }

840 int bluestore_blob_t::verify_csum(uint64_t b_off, const bufferlist& bl,
841  int* b_bad_off, uint64_t *bad_csum) const
842 {
843  int r = 0;
844 
845  *b_bad_off = -1;
846  switch (csum_type) {
847  case Checksummer::CSUM_NONE:
848  break;
849  case Checksummer::CSUM_XXHASH32:
850  *b_bad_off = Checksummer::verify<Checksummer::xxhash32>(
851  get_csum_chunk_size(), b_off, bl.length(), bl, csum_data, bad_csum);
852  break;
853  case Checksummer::CSUM_XXHASH64:
854  *b_bad_off = Checksummer::verify<Checksummer::xxhash64>(
855  get_csum_chunk_size(), b_off, bl.length(), bl, csum_data, bad_csum);
856  break;
857  case Checksummer::CSUM_CRC32C:
858  *b_bad_off = Checksummer::verify<Checksummer::crc32c>(
859  get_csum_chunk_size(), b_off, bl.length(), bl, csum_data, bad_csum);
860  break;
861  case Checksummer::CSUM_CRC32C_16:
862  *b_bad_off = Checksummer::verify<Checksummer::crc32c_16>(
863  get_csum_chunk_size(), b_off, bl.length(), bl, csum_data, bad_csum);
864  break;
865  case Checksummer::CSUM_CRC32C_8:
866  *b_bad_off = Checksummer::verify<Checksummer::crc32c_8>(
867  get_csum_chunk_size(), b_off, bl.length(), bl, csum_data, bad_csum);
868  break;
869  default:
870  r = -EOPNOTSUPP;
871  break;
872  }
873 
874  if (r < 0)
875  return r;
876  else if (*b_bad_off >= 0)
877  return -1; // bad checksum
878  else
879  return 0;
880 }

882 void bluestore_blob_t::allocated(uint32_t b_off, uint32_t length, const PExtentVector& allocs)
883 {
884  if (extents.size() == 0) {
885  // if blob is compressed then logical length to be already configured
886  // otherwise - to be unset.
887  ceph_assert((is_compressed() && logical_length != 0) ||
888  (!is_compressed() && logical_length == 0));
889 
890  extents.reserve(allocs.size() + (b_off ? 1 : 0));
891  if (b_off) {
892  extents.emplace_back(
893  bluestore_pextent_t(bluestore_pextent_t::INVALID_OFFSET, b_off));
894 
895  }
896  uint32_t new_len = b_off;
897  for (auto& a : allocs) {
898  extents.emplace_back(a.offset, a.length);
899  new_len += a.length;
900  }
901  if (!is_compressed()) {
902  logical_length = new_len;
903  }
904  } else {
905  ceph_assert(!is_compressed()); // partial allocations are forbidden when
906  // compressed
907  ceph_assert(b_off < logical_length);
908  uint32_t cur_offs = 0;
909  auto start_it = extents.begin();
910  size_t pos = 0;
911  while (true) {
912  ceph_assert(start_it != extents.end());
913  if (cur_offs + start_it->length > b_off) {
914  break;
915  }
916  cur_offs += start_it->length;
917  ++start_it;
918  ++pos;
919  }
920  uint32_t head = b_off - cur_offs;
921  uint32_t end_off = b_off + length;
922  auto end_it = start_it;
923 
924  while (true) {
925  ceph_assert(end_it != extents.end());
926  ceph_assert(!end_it->is_valid());
927  if (cur_offs + end_it->length >= end_off) {
928  break;
929  }
930  cur_offs += end_it->length;
931  ++end_it;
932  }
933  ceph_assert(cur_offs + end_it->length >= end_off);
934  uint32_t tail = cur_offs + end_it->length - end_off;
935 
936  start_it = extents.erase(start_it, end_it + 1);
937  size_t count = allocs.size();
938  count += head ? 1 : 0;
939  count += tail ? 1 : 0;
940  extents.insert(start_it,
941  count,
942  bluestore_pextent_t(
943  bluestore_pextent_t::INVALID_OFFSET, 0));
944 
945  // Workaround to resolve lack of proper iterator return in vector::insert
946  // Looks like some gcc/stl implementations still lack it despite c++11
947  // support claim
948  start_it = extents.begin() + pos;
949 
950  if (head) {
951  start_it->length = head;
952  ++start_it;
953  }
954  for(auto& e : allocs) {
955  *start_it = e;
956  ++start_it;
957  }
958  if (tail) {
959  start_it->length = tail;
960  }
961  }
962 }

964 // cut it out of extents
965 struct vecbuilder {
966  PExtentVector v;
967  uint64_t invalid = 0;
968 
969  void add_invalid(uint64_t length) {
970  invalid += length;
971  }
972  void flush() {
973  if (invalid) {
974  v.emplace_back(bluestore_pextent_t(bluestore_pextent_t::INVALID_OFFSET,
975  invalid));
976 
977  invalid = 0;
978  }
979  }
980  void add(uint64_t offset, uint64_t length) {
981  if (offset == bluestore_pextent_t::INVALID_OFFSET) {
982  add_invalid(length);
983  }
984  else {
985  flush();
986  v.emplace_back(offset, length);
987  }
988  }
989 };

999 bool bluestore_blob_t::release_extents(bool all,
1000  const PExtentVector& logical,
1001  PExtentVector* r)
1002 {
1003  // common case: all of it?
1004  if (all) {
1005  uint64_t pos = 0;
1006  for (auto& e : extents) {
1007  if (e.is_valid()) {
1008  r->push_back(e);
1009  }
1010  pos += e.length;
1011  }
1012  ceph_assert(is_compressed() || get_logical_length() == pos);
1013  extents.resize(1);
1014  extents[0].offset = bluestore_pextent_t::INVALID_OFFSET;
1015  extents[0].length = pos;
1016  return true;
1017  }
1018  // remove from pextents according to logical release list
1019  vecbuilder vb;
1020  auto loffs_it = logical.begin();
1021  auto lend = logical.end();
1022  uint32_t pext_loffs_start = 0; //starting loffset of the current pextent
1023  uint32_t pext_loffs = 0; //current loffset
1024  auto pext_it = extents.begin();
1025  auto pext_end = extents.end();
1026  while (pext_it != pext_end) {
1027  if (loffs_it == lend ||
1028  pext_loffs_start + pext_it->length <= loffs_it->offset) {
1029  int delta0 = pext_loffs - pext_loffs_start;
1030  ceph_assert(delta0 >= 0);
1031  if ((uint32_t)delta0 < pext_it->length) {
1032  vb.add(pext_it->offset + delta0, pext_it->length - delta0);
1033  }
1034  pext_loffs_start += pext_it->length;
1035  pext_loffs = pext_loffs_start;
1036  ++pext_it;
1037  }
1038  else {
1039  //assert(pext_loffs == pext_loffs_start);
1040  int delta0 = pext_loffs - pext_loffs_start;
1041  ceph_assert(delta0 >= 0);
1042 
1043  int delta = loffs_it->offset - pext_loffs;
1044  ceph_assert(delta >= 0);
1045  if (delta > 0) {
1046  vb.add(pext_it->offset + delta0, delta);
1047  pext_loffs += delta;
1048  }
1049 
1050  PExtentVector::iterator last_r = r->end();
1051  if (r->begin() != last_r) {
1052  --last_r;
1053  }
1054  uint32_t to_release = loffs_it->length;
1055  do {
1056  uint32_t to_release_part =
1057  std::min(pext_it->length - delta0 - delta, to_release);
1058  auto o = pext_it->offset + delta0 + delta;
1059  if (last_r != r->end() && last_r->offset + last_r->length == o) {
1060  last_r->length += to_release_part;
1061  }
1062  else {
1063  last_r = r->emplace(r->end(), o, to_release_part);
1064  }
1065  to_release -= to_release_part;
1066  pext_loffs += to_release_part;
1067  if (pext_loffs == pext_loffs_start + pext_it->length) {
1068  pext_loffs_start += pext_it->length;
1069  pext_loffs = pext_loffs_start;
1070  pext_it++;
1071  delta0 = delta = 0;
1072  }
1073  } while (to_release > 0 && pext_it != pext_end);
1074  vb.add_invalid(loffs_it->length - to_release);
1075  ++loffs_it;
1076  }
1077  }
1078  vb.flush();
1079  extents.swap(vb.v);
1080  return false;
1081 }

1083 void bluestore_blob_t::split(uint32_t blob_offset, bluestore_blob_t& rb)
1084 {
1085  size_t left = blob_offset;
1086  uint32_t llen_lb = 0;
1087  uint32_t llen_rb = 0;
1088  unsigned i = 0;
1089  for (auto p = extents.begin(); p != extents.end(); ++p, ++i) {
1090  if (p->length <= left) {
1091  left -= p->length;
1092  llen_lb += p->length;
1093  continue;
1094  }
1095  if (left) {
1096  if (p->is_valid()) {
1097  rb.extents.emplace_back(bluestore_pextent_t(p->offset + left,
1098  p->length - left));
1099  }
1100  else {
1101  rb.extents.emplace_back(bluestore_pextent_t(
1102  bluestore_pextent_t::INVALID_OFFSET,
1103  p->length - left));
1104  }
1105  llen_rb += p->length - left;
1106  llen_lb += left;
1107  p->length = left;
1108  ++i;
1109  ++p;
1110  }
1111  while (p != extents.end()) {
1112  llen_rb += p->length;
1113  rb.extents.push_back(*p++);
1114  }
1115  extents.resize(i);
1116  logical_length = llen_lb;
1117  rb.logical_length = llen_rb;
1118  break;
1119  }
1120  rb.flags = flags;
1121 
1122  if (has_csum()) {
1123  rb.csum_type = csum_type;
1124  rb.csum_chunk_order = csum_chunk_order;
1125  size_t csum_order = get_csum_chunk_size();
1126  ceph_assert(blob_offset % csum_order == 0);
1127  size_t pos = (blob_offset / csum_order) * get_csum_value_size();
1128  // deep copy csum data
1129  bufferptr old;
1130  old.swap(csum_data);
1131  rb.csum_data = bufferptr(old.c_str() + pos, old.length() - pos);
1132  csum_data = bufferptr(old.c_str(), pos);
1133  }
1134 }
