381  void _add_buffer(BufferCacheShard* cache, Buffer* b, int level, Buffer* near) {
382  cache->_audit("_add_buffer start");
383  buffer_map[b->offset].reset(b);
384  if (b->is_writing()) {
385  // we might get already cached data for which resetting mempool is inppropriate
386  // hence calling try_assign_to_mempool
387  b->data.try_assign_to_mempool(mempool::mempool_bluestore_writing);
388  if (writing.empty() || writing.rbegin()->seq <= b->seq) {
389  writing.push_back(*b);
390  } else {
391  auto it = writing.begin();
392  while (it->seq < b->seq) {
393  ++it;
394  }
395 
396  ceph_assert(it->seq >= b->seq);
397  // note that this will insert b before it
398  // hence the order is maintained
399  writing.insert(it, *b);
400  }
401  } else {
402  b->data.reassign_to_mempool(mempool::mempool_bluestore_cache_data);
403  cache->_add(b, level, near);
404  }
405  cache->_audit("_add_buffer end");
406  }
407  void _rm_buffer(BufferCacheShard* cache, Buffer *b) {
408  _rm_buffer(cache, buffer_map.find(b->offset));
409  }
410  std::map<uint32_t, std::unique_ptr<Buffer>>::iterator
411  _rm_buffer(BufferCacheShard* cache,
412  std::map<uint32_t, std::unique_ptr<Buffer>>::iterator p) {
413  ceph_assert(p != buffer_map.end());
414  cache->_audit("_rm_buffer start");
415  if (p->second->is_writing()) {
416  writing.erase(writing.iterator_to(*p->second));
417  } else {
418  cache->_rm(p->second.get());
419  }
420  p = buffer_map.erase(p);
421  cache->_audit("_rm_buffer end");
422  return p;
423  }

425  std::map<uint32_t,std::unique_ptr<Buffer>>::iterator _data_lower_bound(
426  uint32_t offset) {
427  auto i = buffer_map.lower_bound(offset);
428  if (i != buffer_map.begin()) {
429  --i;
430  if (i->first + i->second->length <= offset)
431  ++i;
432  }
433  return i;
434  }

550  struct SharedBlobSet {
552  ceph::mutex lock = ceph::make_mutex("BlueStore::SharedBlobSet::lock");
553 
554  // we use a bare pointer because we don't want to affect the ref
555  // count
556  mempool::bluestore_cache_meta::unordered_map<uint64_t,SharedBlob*> sb_map;
557 
558  SharedBlobRef lookup(uint64_t sbid) {
559  std::lock_guard l(lock);
560  auto p = sb_map.find(sbid);
561  if (p == sb_map.end() ||
562  p->second->nref == 0) {
563  return nullptr;
564  }
565  return p->second;
566  }
567 
568  void add(Collection* coll, SharedBlob *sb) {
569  std::lock_guard l(lock);
570  sb_map[sb->get_sbid()] = sb;
571  sb->coll = coll;
572  }
573 
574  bool remove(SharedBlob *sb, bool verify_nref_is_zero=false) {
575  std::lock_guard l(lock);
576  ceph_assert(sb->get_parent() == this);
577  if (verify_nref_is_zero && sb->nref != 0) {
578  return false;
579  }
580  // only remove if it still points to us
581  auto p = sb_map.find(sb->get_sbid());
582  if (p != sb_map.end() &&
583  p->second == sb) {
584  sb_map.erase(p);
585  }
586  return true;
587  }
588 
589  bool empty() {
590  std::lock_guard l(lock);
591  return sb_map.empty();
592  }

1041  int seek_shard(uint32_t offset) {
1042  size_t end = shards.size();
1043  size_t mid, left = 0;
1044  size_t right = end; // one passed the right end
1045 
1046  while (left < right) {
1047  mid = left + (right - left) / 2;
1048  if (offset >= shards[mid].shard_info->offset) {
1049  size_t next = mid + 1;
1050  if (next >= end || offset < shards[next].shard_info->offset)
1051  return mid;
1052  //continue to search forwards
1053  left = next;
1054  } else {
1055  //continue to search backwards
1056  right = mid;
1057  }
1058  }
1059 
1060  return -1; // not found
1061  }

1148  /*
1149  The primary idea of the collector is to estimate a difference between
1150  allocation units(AU) currently present for compressed blobs and new AUs
1151  required to store that data uncompressed.
1152  Estimation is performed for protrusive extents within a logical range
1153  determined by a concatenation of old_extents collection and specific(current)
1154  write request.
1155  The root cause for old_extents use is the need to handle blob ref counts
1156  properly. Old extents still hold blob refs and hence we need to traverse
1157  the collection to determine if blob to be released.
1158  Protrusive extents are extents that fit into the blob std::set in action
1159  (ones that are below the logical range from above) but not removed totally
1160  due to the current write.
1161  E.g. for
1162  extent1 <loffs = 100, boffs = 100, len = 100> ->
1163  blob1<compressed, len_on_disk=4096, logical_len=8192>
1164  extent2 <loffs = 200, boffs = 200, len = 100> ->
1165  blob2<raw, len_on_disk=4096, llen=4096>
1166  extent3 <loffs = 300, boffs = 300, len = 100> ->
1167  blob1<compressed, len_on_disk=4096, llen=8192>
1168  extent4 <loffs = 4096, boffs = 0, len = 100> ->
1169  blob3<raw, len_on_disk=4096, llen=4096>
1170  write(300~100)
1171  protrusive extents are within the following ranges <0~300, 400~8192-400>
1172  In this case existing AUs that might be removed due to GC (i.e. blob1)
1173  use 2x4K bytes.
1174  And new AUs expected after GC = 0 since extent1 to be merged into blob2.
1175  Hence we should do a collect.
1176  */
